Basic :

```jsx
<Tooltip content="I am at the top Yo" id="topTooltip">
  <Button>Top</Button>
</Tooltip>
```

```jsx
<Tooltip
  open
  placement="bottom"
  content="But now I am at the bottom Yo"
  id="bottomTooltip"
>
  <Button>Bottom</Button>
</Tooltip>
```

```jsx
<Tooltip
  open
  placement="right"
  content="I'm right Yo"
  id="rightTooltip"
  use="Info"
>
  <Button>Right</Button>
</Tooltip>
```

```jsx
<Tooltip
  open
  placement="left"
  content="Left wins the day tho !"
  id="leftTooltip"
  use="Error"
>
  <Button>Left</Button>
</Tooltip>
```
